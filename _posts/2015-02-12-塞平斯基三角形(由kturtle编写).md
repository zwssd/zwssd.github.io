#kturtle-script-v1.0
# Sierpinski triangle
reset
go 100, 100
print "塞平斯基三角形"
canvassize 800, 600
go 100, 500
direction 90
learn sierp $length {
  if $length > 2 {
    repeat 3 {
      sierp $length/2
      forward $length
      turnleft 120
    }
  }
}sierp 500
