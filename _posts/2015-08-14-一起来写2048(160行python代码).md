def move(mtr, dirct):
  score = 0
  visit = []
  if dirct == 0:  # left
    for i in range(4):
      for j in range(1, 4):
        for k in range(j,0,-1):
          if mtr[i][k - 1] == 0 :
            mtr[i][k - 1] = mtr[i][k]
            mtr[i][k] = 0
          elif mtr[i][k - 1] == mtr[i][k] and 4 * i + k - 1 not in visit and 4 * i + k not in visit:
            mtr[i][k - 1] *= 2
            mtr[i][k] = 0
            score += mtr[i][k - 1]
            visit.append(4 * i + k)
            visit.append(4 * i + k - 1)
    # for i in range(4):
    #	for j in range(3):
  elif dirct == 1:  # down
    for j in range(4):
      for i in range(3, 0, -1):
        for k in range(0,i):
           if mtr[k+1][j] == 0:
            mtr[k+1][j] = mtr[k][j]
            mtr[k][j]=0
           elif mtr[k+1][j]==mtr[k][j] and (4 *(k+1)+j) not in visit and (4*k+j) not in visit:
            mtr[k+1][j]*=2
            mtr[k][j]=0
            score=mtr[k+1][j]
            visit.append(4*(k)+j)
            visit.append(4*(k+1)+j)
  elif dirct == 2:  # up
    for j in range(4):
      for i in range(1,4):
        for k in range(i,0,-1):
          if mtr[k-1][j]==0:
            mtr[k-1][j]=mtr[k][j]
            mtr[k][j]=0
          elif mtr[k-1][j]==mtr[k][j] and (4 *(k-1)+j) not in visit and (4*k+j) not in visit:
            mtr[k-1][j]*=2
            mtr[k][j]=0
            score += mtr[k-1][j]
            visit.append(4*(k)+j)
            visit.append(4*(k-1)+j)
  elif dirct == 3:  # right
    for i in range(4):
      for j in range(3, 0, -1):
        for k in range(j):
           if mtr[i][k+1]  == 0:
            mtr[i][k+1] = mtr[i][k]
            mtr[i][k]=0
           elif mtr[i][k] ==mtr[i][k+1] and 4 * i + k + 1 not in visit and 4 * i + k not in visit:
            mtr[i][k+1]*=2
            mtr[i][k]=0
            score+=mtr[i][k+1]
            visit.append(4*i+k+1)
            visit.append(4*i+k)
  return score
